# M7 Part A — Admin UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the 14-page identity admin surface defined in spec §8, gated behind a working `useIdentity()` hook and `<RequirePermission>` guard, so `ENABLE_IDENTITY=true` deployments can log in via OIDC and administer tenants/users/workspaces/tokens/audit without touching the DB.

**Architecture:**
- Single Next.js 16 app, route groups `(public)` for login/callback/logout and `(admin)` for admin/*; existing chat UI lives under `(app)` (to be added in a later PR — out of scope here).
- Client-side identity cache via TanStack Query hitting `/api/me`; server-side guard via `middleware.ts` redirect-on-no-cookie.
- Each admin page calls typed fetch wrappers in `src/core/identity/api.ts` that forward credentials (`credentials: "include"`) and bubble 401 to a global session-expired modal.
- Backend gets a small sidecar router (`app/gateway/identity/routers/admin.py`) that fills the admin/tenant-admin endpoints from spec §8.4 currently missing (tenants list/create/detail, users list/create/detail, workspaces list/create/detail + members, tenant-wide tokens). Read-heavy first; create/mutate minimal.

**Tech Stack:** Next.js 16 App Router (React 19), TypeScript 5.8, Tailwind 4, shadcn/ui + Radix primitives, TanStack Query v5, React Hook Form + Zod (new), Playwright, Vitest, FastAPI (Python 3.12+).

**Sequencing — land as four PRs under branch `feat/m7-admin-ui`:**

| PR | Scope | Pages | Depends on |
|----|-------|-------|-----------|
| **A1 — Foundation** | identity core + middleware + login + logout + callback + profile + 403 + session-expired modal + Playwright harness + 2 E2E | 1-4, 14 (5 of 14) | none |

```typescript
// frontend/src/core/identity/fetcher.ts
import { type IdentityError } from "./types";

type Listener = () => void;
const listeners = new Set<Listener>();

export function onSessionExpired(fn: Listener): () => void {
  listeners.add(fn);
  return () => listeners.delete(fn);
}

export function resetSessionExpiredListeners(): void {
  listeners.clear();
}

function emitSessionExpired(): void {
  for (const fn of listeners) fn();
}

export async function identityFetch<T>(
  input: string,
  init?: RequestInit,
): Promise<T> {
  const resp = await fetch(input, {
    credentials: "include",
    headers: {
      accept: "application/json",
      ...(init?.body ? { "content-type": "application/json" } : {}),
      ...init?.headers,
    },
    ...init,
  });

  if (resp.status === 401) {
    emitSessionExpired();
    const err: IdentityError = { kind: "unauthenticated" };
    throw err;
  }
  if (resp.status === 403) {
    let missing: string | undefined;
    try {
      const body = (await resp.json()) as { detail?: { missing?: string } };
      missing = body?.detail?.missing;
    } catch {
      // 403 without JSON body is valid; missing stays undefined.
    }
    const err: IdentityError = { kind: "forbidden", missing };
    throw err;
  }
  if (!resp.ok) {
    const text = await resp.text().catch(() => "");
    const err: IdentityError = {
      kind: "network",
      status: resp.status,
      message: text,
    };
    throw err;
  }

  return (await resp.json()) as T;
}
```

```typescript
// frontend/src/core/identity/hooks.ts
"use client";

import { useMutation, useQuery, useQueryClient } from "@tanstack/react-query";
import { useMemo } from "react";

import { identityApi } from "./api";
import { identityKeys } from "./query-keys";
import { type IdentityError, type MeResponse, type Permission } from "./types";

export function useIdentity() {
  const query = useQuery<MeResponse, IdentityError>({
    queryKey: identityKeys.me(),
    queryFn: identityApi.me,
    retry: false,
    staleTime: 60_000,
  });

  return {
    identity: query.data,
    isLoading: query.isLoading,
    isAuthenticated: query.isSuccess && !!query.data,
    error: query.error,
    refetch: query.refetch,
  };
}

export function useHasPermission(perm: Permission): boolean {
  const { identity } = useIdentity();
  return useMemo(
    () => !!identity?.permissions.includes(perm),
```

### Task A1.12: Frontend — `middleware.ts` guard

**Why:** Spec §8.2 requires server-side redirect for unauthenticated `/admin/*`. Using the session cookie's presence as the signal is fine — the backend still enforces on every API call, so a forged cookie can't reach data.

```typescript
// frontend/middleware.ts
import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";

// Default cookie name — override with NEXT_PUBLIC_DEERFLOW_COOKIE_NAME if backend
// is configured with DEERFLOW_COOKIE_NAME=<custom>.
const COOKIE_NAME =
  process.env.NEXT_PUBLIC_DEERFLOW_COOKIE_NAME ?? "deerflow_session";

export function middleware(req: NextRequest) {
  const session = req.cookies.get(COOKIE_NAME);
  if (session?.value) {
    return NextResponse.next();
  }

  const url = req.nextUrl.clone();
  const next = req.nextUrl.pathname + req.nextUrl.search;
  url.pathname = "/login";
  url.search = `?next=${encodeURIComponent(next)}`;
  return NextResponse.redirect(url);
}

export const config = {
  matcher: ["/admin/:path*"],
};
```

```typescript
import { expect, test } from "@playwright/test";
```

```typescript
test("session-expired modal appears when /api/me returns 401 mid-session", async ({
  page,
}) => {
  let firstCall = true;

  await page.route("**/api/me", (route) => {
    if (firstCall) {
      firstCall = false;
      return route.fulfill({
        status: 200,
        contentType: "application/json",
        body: JSON.stringify({
          user_id: 1,
          email: "a@b",
          display_name: "A",
          avatar_url: null,
          active_tenant_id: 1,

          workspaces: [{ id: 7, slug: "main", name: "Main" }],
          permissions: ["tenant:read"],
          roles: {},
        }),
      });
    }
    return route.fulfill({ status: 401, body: "" });
  });
  await mockIdentity(page, { authenticated: true });

  await page.goto("/admin/profile");
  await expect(page.getByRole("heading", { name: "A" })).toBeVisible();

  // Force a refetch to trigger the second (401) response.
  await page.evaluate(() => {
    void fetch("/api/me", { credentials: "include" });
  });

  await expect(

  ).toBeVisible();
  await expect(
    page.getByRole("link", { name: /go to sign-in/i }),
  ).toBeVisible();
});
```
